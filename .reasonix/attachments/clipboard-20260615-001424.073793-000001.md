## 两版统一同步日志格式

手机版使用以下统一的日志消息格式。手表版照此实现，两边日志一模一样，排障时可以直接对比。

### 1. 通用方法

```javascript
var _syncLogs = [];                    // 存储所有日志行
var _syncLogMaxSummary = 6;           // 收起时显示的行数

// 添加一行日志
function syncDebug(msg) {
  _syncLogs.push(msg);
  renderSyncDebug();  // 刷新 UI
}

// 清空日志
function syncDebugClear() {
  _syncLogs = [];
  // 隐藏日志面板
}

// 渲染日志面板（可展开/收起）
function renderSyncDebug() { /* 见下方 UI 实现 */ }
function toggleSyncDebug() { /* 展开/收起切换 */ }
```

### 2. 每条日志消息格式（照抄）

| 场景 | 日志消息 | 说明 |
|------|---------|------|
| 同步开始 | `🔄 开始同步...` | — |
| 用户标识 | `用户 ID: ad639ebb...` | 取 `user_id` 前 8 位 |
| 本地状态 | `📱 手机本地: 3 条, 1250ml` | 设备本地的今日记录数和总水量 |
| 推送开始 | `📤 推送 3 条, 总量 1250ml, 连续 2 天` | 推送的记录数、总水量、连续达标天数 |
| 推送 ID | `📤 推送记录 ID: abc123, def456` | 每条记录 ID 的前 6 位，逗号分隔 |
| 推送成功 | `✅ 推送成功` | — |
| 推送失败 | `❌ 推送失败: 错误信息` | 显示具体错误 |
| 拉取开始 | `📥 拉取云端记录合并到本地...` | — |
| 拉取请求 | `🔍 GET https://xxx.supabase.co/rest/v1/...` | 实际请求的 URL（前 60 字符） |
| 拉取状态 | `🔍 HTTP 200 OK` | HTTP 状态码 |
| 拉取结果 | `🔍 返回 today_records: 3 条` | 云端返回的记录数 |
| 拉取详情 | `🔍 记录详情: abc123=500ml😊, def456=1000ml` | 每条记录的 ID 前 6 位 + 水量 + 表情 |
| 拉取空 | `🔍 返回空，无云端记录` | 云端无数据 |
| 拉取失败 | `🔍 请求失败` 或 `🔍 网络异常: ...` | 网络错误时 |
| 合并对比 | `📥 本地已有 2 条, 云端 3 条` | 合并前本地和云端各自的数量 |
| 合并结果 | `📥 合并了 1 条: xxxyyy` | 实际新增的记录数和 ID |
| 合并无新增 | `📥 无新记录（云端 ID 全部已在本地）` | 没有需要合并的新记录 |
| 同步完成 | `✅ 同步完成` | — |
| 同步后状态 | `📱 手机本地现在: 3 条, 1750ml` | 同步后的本地记录数和总水量 |
| 检测到手表数据 | `⌚ 本次合并了 2 条手表记录 ✅` | 从云端合并过来的记录数 |
| 未检测到手表 | `⌚ 未检测到手表的记录 ❌` | 云端与本地一致，无手表新增数据 |
| 同步失败 | `❌ 同步失败: 错误信息` | catch 到的异常 |

### 3. 日志面板 UI

**HTML：**
```html
<div id="syncDebugWrap" style="display:none;margin:6px 0;">
  <!-- 摘要区：收起时可见，可点击展开 -->
  <div id="syncDebugSummary" style="font-size:0.75rem;color:#666;
       background:#f8f9fa;border-radius:8px;padding:8px 10px;
       line-height:1.6;cursor:pointer;" onclick="toggleSyncDebug()"></div>
  <!-- 完整区：展开时可见，可滚动 -->
  <div id="syncDebugFull" style="display:none;font-size:0.6875rem;color:#555;
       background:#f0f2f5;border-radius:8px;padding:8px 10px;
       line-height:1.5;max-height:300px;overflow-y:auto;
       word-break:break-all;"></div>
</div>
```

**展开/收起渲染逻辑：**
```javascript
function renderSyncDebug() {
  var wrap = document.getElementById('syncDebugWrap');
  var summary = document.getElementById('syncDebugSummary');
  var full = document.getElementById('syncDebugFull');
  if (!wrap || !summary || !full) return;
  wrap.style.display = 'block';

  var isExpanded = full.style.display === 'block';
  if (isExpanded) {
    // 展开：隐藏摘要区，完整区显示全部
    summary.style.display = 'none';
    var fullHtml = '';
    for (var i = 0; i < _syncLogs.length; i++) {
      fullHtml += '<div>' + _syncLogs[i] + '</div>';
    }
    fullHtml += '<div style="color:#999;margin-top:4px;cursor:pointer;text-align:center;" onclick="toggleSyncDebug()">▲ 收起</div>';
    full.innerHTML = fullHtml;
    full.scrollTop = full.scrollHeight;
    full.style.display = 'block';
  } else {
    // 收起：显示摘要前 6 行，隐藏完整区
    summary.style.display = 'block';
    full.style.display = 'none';
    var showLines = Math.min(_syncLogs.length, 6);
    var summaryHtml = '';
    for (var i = 0; i < showLines; i++) {
      summaryHtml += '<div>' + _syncLogs[i] + '</div>';
    }
    if (_syncLogs.length > 6) {
      summaryHtml += '<div style="color:#4A90E2;font-weight:600;margin-top:2px;">▼ 展开全部 (' + _syncLogs.length + ' 条)</div>';
    }
    summary.innerHTML = summaryHtml;
  }
}

function toggleSyncDebug() {
  var full = document.getElementById('syncDebugFull');
  if (!full) return;
  full.style.display = full.style.display === 'block' ? 'none' : 'block';
  renderSyncDebug();
}
```

### 4. 调用位置

在 `syncTodayToLeaderboard()` 和 `mergeCloudTodayRecords()` 以及 `forceSync()` 中按第 2 节的格式添加 `syncDebug(...)` 调用。
